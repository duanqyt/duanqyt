#include <iostream>
using namespace std;
char upper_E3(char str,int shift)
{
        return char((str-65+shift)%26+65);
}
char lower_E3(char str,int shift)
{
        return char((str-97+shift)%26+97);
}
int main()
{
    string str ="";//明文
    string str1="";//密文
    int shift = 3;
    cout << "输入明文：";
    getline(cin,str);
    cout << "输入移位量：";
    cin >> shift;
    for(int i =0;i<str.length();i++)
    {
        if(isalpha(str[i]))
        {
            if(islower(str[i]))
                str1 += lower_E3(str[i],shift);
            if(isupper(str[i]))
                str1 += upper_E3(str[i],shift);
        }
        else
        {
            str1 += str[i];
        }
    }
    cout << "密文：" <<str1 <<endl;
    return 0;
}
