//初始化表
Status InitList(SqList& L)
{
    L.elem = new ElemType[MAXSIZE];
    if (!L.elem) exit(OVERFLOW);
    L.length = 0;
    return OK;
}

//清空
void ClearList(SqList &L)
{
    L.length = 0;
}

//判空
void isEmpty(SqList L)
{
    if (L.length <= 0) {
        cout << "List is empity!" << endl;
    }
    else {
        cout << "List is not empity!" << endl;
    }

}

//表长
void getLength(SqList L)
{
    int length = L.length;
    cout << length << endl;
}

//取指定位置的值
Status GetElem(SqList L, int i, ElemType& e)
{
    //判断i是否合理
    if (i<1 || i>L.length) 
    {
        cout << "位置非法！" << endl;
        return ERROR;
    }
    e = L.elem[i-1];
    return OK;
}

//求前驱
Status getPrecursor(SqList L, int i, ElemType& e)
{
    if ((i <= 1) || (i > L.length))
    {
        cout << "前驱不存在" << endl;
        return ERROR;
    }

    e = L.elem[i-2];
    return OK;
}

//求后继
Status getSuccessor(SqList L, int i, ElemType& e)
{
    if ((i < 1) || (i > L.length - 1))
    {
        cout << "后继不存在" << endl;
        return ERROR;
    }
    e = L.elem[i];
    return OK;
}

//插入
Status ListInsert(SqList &L, int i,ElemType e)
{
    if ((i < 1) || (i > L.length + 1))
    {
        cout << "插入失败！" << endl;
        return ERROR;
    }
    if (L.length == MAXSIZE) return ERROR;
    for (int j = L.length-1; j >= i-1; j--)
    {
        L.elem[j + 1] = L.elem[j];
    }
    L.elem[i - 1] = e;
    ++L.length;
    cout << "插入成功！" << endl;
    return OK;
}

//删除线性表指定位置元素
Status ListDelete(SqList &L,int i)
{
    if ((i < 1) || (i > L.length))
    {
        cout << "删除失败！" << endl;
        return ERROR;
    }
    for (int j = i; j<=L.length-1; j++)
    {
        L.elem[j - 1] = L.elem[j];
    }
    L.length--;
    cout << "删除成功！" << endl;
    return OK;
}
//9----显示线性表
void display(SqList L)
{
    if (L.length > 0)
        for (int j = 0; j < L.length; )
        {
            cout << L.elem[j] << "  ";
            ++j;
            if (j % 10 == 0)
            {
                cout << endl;
            }
        }
    else
        cout << "空表" << endl; 
}
