a,b,c=map(int,input().split())
def duan(a,b,c):
    n=1
    while True:
        if n%a==0 and n%b==0 and n%c==0:
            return n
        else:
            n+=1
print(duan(a,b,c))

lst=[]
n = eval(input())
def isPrime(n):
    for i in range(2,int(pow(n,0.5))+1):
        if n%i==0:
            return False
    return True
start=2
end=n
while True:
    if n%start==0 and isPrime(start):
        lst.append(start)
        n/=start
    if n%start!=0:
        start+=1
    if start==end:
        break
for i in range(0,len(lst)):
    if i==(len(lst)-1):
        print(lst[i],end="")
    else:
        print(lst[i],end=" ")

153是一个非常特殊的数，它等于它的每位数字的立方和，即153=111+555+333。编程求所有满足这种条件的三位十进制数。

def isSpecialNum(n):
    sum = 0
    a=n//100
    b=n%100//10
    c=n%10
    sum = sum + a*a*a + b*b*b+c*c*c
    if sum == n:
        return True
    else:
        return False
for i in range(100,1000):
    if isSpecialNum(i):
        print(i)
